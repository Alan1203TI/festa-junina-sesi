Cole estas regras no Firebase > Firestore Database > Regras > Publicar:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }
    function isAdmin() {
      return request.auth != null && request.auth.token.email == 'admin@festajunina.com';
    }

    match /produtos/{document=**} {
      allow read: if signedIn();
      allow create, update, delete: if isAdmin();
    }

    match /vendas/{document=**} {
      allow create, read: if signedIn();
      allow update, delete: if isAdmin();
    }
  }
}
